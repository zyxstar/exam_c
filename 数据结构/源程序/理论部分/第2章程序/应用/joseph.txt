#include <stdlib.h>
#include <alloc.h>
typedef struct node
{
    int number; /* 人的序号 */
    int cipher; /* 密码 */
    struct node *next; /* 指向下一个节点的指针 */
}jnode;

jnode *CreatList(int num) /* 建立循环链表 */
{
    int i;
    jnode *ptr1,*head;

    if((ptr1=(jnode *)malloc(sizeof(jnode)))==NULL)
    {
        perror("malloc");
        return ptr1;
    }
    head=ptr1;
    ptr1->next=head;
    for(i=1;i<num;i++)
    {
        if((ptr1->next=(struct node *)malloc(sizeof(struct node)))==NULL)
        {
            perror("malloc");
            ptr1->next=head;
            return head;
        }
        ptr1=ptr1->next;
        ptr1->next=head;
    }
    return head;
}
